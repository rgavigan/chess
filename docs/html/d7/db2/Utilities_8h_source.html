187inline std::vector<std::string> parseStockfishOutput(const std::string& output) {
188 std::vector<std::string> suggestedMoves;
189
190 // Use regular expressions to match lines containing suggested moves, using a-h as columns as the first and third characters and 1-8 as rows as the second and fourth characters
191 std::regex movePattern(R"(\b[a-h]\d[a-h]\d\b)");
192
193 // Split the output into lines using the split function
194 std::vector<std::string> lines = splitString(output, '\n');
195
196 for (const std::string& line : lines) {
197 std::smatch match;
198 if (std::regex_search(line, match, movePattern)) {
199 for (const auto& m : match) {
200 suggestedMoves.push_back(m.str());
201 }
202 }
203 }
204
205 return suggestedMoves;
206}

296inline std::string generatePGN(
297 const std::string& eventName,
298 const std::string& site,
299 const std::string& date,
300 const std::string& round,
301 const std::string& whitePlayer,
302 const std::string& blackPlayer,
303 const std::string& result,
304 const std::vector<BoardMetadata>& chronologicalData
305) {
306 std::stringstream pgn;
307
308 // Add meta information
309 pgn << "[Event \"" << eventName << "\"]\n"
310 << "[Site \"" << site << "\"]\n"
311 << "[Date \"" << date << "\"]\n"
312 << "[Round \"" << round << "\"]\n"
313 << "[White \"" << whitePlayer << "\"]\n"
314 << "[Black \"" << blackPlayer << "\"]\n"
315 << "[Result \"" << result << "\"]\n\n";
316
317 for (const auto& metadata : chronologicalData) {
318 const Move& move = metadata.moveMade;
319 std::cout << move.start.col << "," << move.start.row << " " << move.end.col << "," << move.end.row << std::endl;
320 if (move.start == Position{0,0} && move.end == Position{0,0}) {
321 continue;
322 }
323
324 // Adding turn number for white moves
325 if (move.playerColour == Colour::WHITE) {
326 pgn << metadata.turnNumber << ". ";
327 }
328
329 // Check for castling
330 if (move.pieceMoved == PieceType::KING && abs(move.end.col - move.start.col) == 2) {
331 pgn << (move.end.col > move.start.col ? "O-O" : "O-O-O");
332 } else {
333 pgn << pieceTypeToNotation(move.pieceMoved);
334
335 // For pawns capturing, we include the starting file
336 if (move.pieceMoved == PieceType::PAWN && move.pieceCaptured != PieceType::NONE) {
337 pgn << static_cast<char>('a' + move.start.col);
338 }
339
340 pgn << (move.pieceCaptured != PieceType::NONE ? "x" : "")
341 << positionToChessNotation(move.end);
342
343 // Pawn promotion
344 if (move.promotionPiece != PieceType::NONE) {
345 pgn << "=" << pieceTypeToNotation(move.promotionPiece);
346 }
347 }
348
349 // Check and Checkmate based on GameStatus from metadata
350 if (metadata.gameStatus == GameStatus::CHECKMATE) {
351 pgn << "#";
352 } else if (metadata.gameStatus == GameStatus::CHECK) {
353 pgn << "+";
354 }
355
356 pgn << " ";
357
358 // Adding newline after every full turn for readability
359 if (move.playerColour == Colour::BLACK || &metadata == &chronologicalData.back()) {
360 pgn << "\n";
361 }
362 }
363
364 // Add the result at the end
365 pgn << result;
366
367 return pgn.str();
368}

370inline bool columnExists(sqlite3* db, const std::string& tableName, const std::string& columnName) {
371 bool exists = false;
372
373 std::string query = "PRAGMA table_info(" + tableName + ");";
374 sqlite3_stmt* stmt;
375
376 if (sqlite3_prepare_v2(db, query.c_str(), -1, &stmt, nullptr) == SQLITE_OK) {
377 while (sqlite3_step(stmt) == SQLITE_ROW) {
378 std::string currentColumnName = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 1));
379 if (currentColumnName == columnName) {
380 exists = true;
381 break;
382 }
383 }
384 }
385
386 sqlite3_finalize(stmt);
387 return exists;
388}
